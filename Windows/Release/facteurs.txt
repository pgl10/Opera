# Ce fichier de commandes facteurs.txt calcule
# les np facteurs premiers p[1], p[2], ... du nombre entier n > 1
# C'est aussi une démonstration de la commande "prochain"
# Exemple avec n=2323 et f=0 : exec facteurs.txt n,f
copier n,p
frac n
1+(last<0)*(last>0)
continuer
nm = n
nm > 1
continuer
0 < 1
boucle
   facteur nm
   p = last
   prochain p
   # si facteur en échec : last = 0
   si last < 1
   quitter
   # si p = nm : c'est le dernier facteur premier
   si p > nm - 1
   quitter
   nm = nm/p
retour
