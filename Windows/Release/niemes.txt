# Des exemples de racines n-ièmes approximées 
setapprox = 100
x1 = 1234567890123456789012345678901234567/77
n1 = 5
r1 = x1^(1/n1)
valeur r1
valeur (x1-r1^n1)/x1
xi = 1/x1
ni = 9
ri = xi^(1/ni)
valeur ri
valeur (xi-ri^ni)/xi
x2 = x1*x1
n2 = 13
r2 = x2^(1/n2)
valeur r2
valeur (x2-r2^n2)/x2
xn = -x2
nn = 17
rn = xn^(1/nn)
valeur rn
valeur (xn-rn^nn)/xn
